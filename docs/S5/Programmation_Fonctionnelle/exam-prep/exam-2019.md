# Examen 2019 -- Expressions lineaires et anneaux

## Partie 1 : Expressions lineaires

### Type de base

```ocaml
type linexpr = {cst : int; coeffs : (int * string) list}
(* Represente : cst + sum(coeff_i * var_i) *)

let e = {cst = 2; coeffs = [(7, "x"); (9, "y")]}
(* Represente : 2 + 7x + 9y *)
```

### Q1 : Evaluation

```ocaml
(* eval_lin : (string -> int) -> linexpr -> int *)
let eval_lin f le =
  let rec eval_li l = match l with
    | [] -> 0
    | (a, b) :: rest -> a * (f b) + eval_li rest
  in
  le.cst + eval_li le.coeffs
```

**Trace** : `eval_lin (fun v -> if v = "x" then 1 else 2) {cst=2; coeffs=[(7,"x");(9,"y")]}`
```
= 2 + eval_li [(7,"x"); (9,"y")]
= 2 + (7 * f("x") + eval_li [(9,"y")])
= 2 + (7 * 1 + (9 * f("y") + 0))
= 2 + (7 + (9 * 2))
= 2 + 7 + 18 = 27
```

### Q2-Q4 : Verification des contraintes

```ocaml
(* C1 : pas de coefficient nul *)
let check1 le =
  let rec checkl1 l = match l with
    | [] -> true
    | (a, _) :: rest -> if a = 0 then false else checkl1 rest
  in checkl1 le.coeffs

(* C2 : variables triees *)
let check2 le =
  let rec checkl2 l = match l with
    | [] | [_] -> true
    | (_, v1) :: ((_, v2) :: _ as rest) ->
        if v1 > v2 then false else checkl2 rest
  in checkl2 le.coeffs

(* C3 : pas de doublons de variables *)
let check3 le =
  let rec checkl3 l = match l with
    | [] -> true
    | (_, v) :: rest ->
        if existe v rest then false else checkl3 rest
  in checkl3 le.coeffs
```

### Q6-Q7 : Constructeurs

```ocaml
let constant c = {cst = c; coeffs = []}
(* constant 5 = {cst = 5; coeffs = []} represente la constante 5 *)

let variable ci xi = {cst = 0; coeffs = [(ci, xi)]}
(* variable 3 "x" = {cst = 0; coeffs = [(3,"x")]} represente 3x *)
```

### Q8 : Multiplication par un scalaire

```ocaml
let mul i le =
  let rec coeffsi l = match l with
    | [] -> []
    | (a, b) :: rest -> (a * i, b) :: coeffsi rest
  in
  {cst = le.cst * i; coeffs = coeffsi le.coeffs}
```

### Q10 : Normalisation

```ocaml
let normalise le =
  if not (check1 le) then normalise (norm1 le)
  else if not (check2 le) then normalise (norm2 le)
  else if not (check3 le) then normalise (norm3 le)
  else le
```

**Principe** : appliquer les normalisations (suppression des zeros, tri, fusion des doublons) jusqu'a ce que les 3 contraintes soient satisfaites.

## Partie 2 : Expressions generales et anneaux

### Types

```ocaml
type 'a expr =
  | Cst of 'a
  | Var of string
  | Add of 'a expr * 'a expr
  | Mul of 'a expr * 'a expr

type 'a anneau = {
  addition : 'a -> 'a -> 'a;
  multiplication : 'a -> 'a -> 'a;
  zero : 'a;
  one : 'a;
  equal : 'a -> 'a -> bool;
}
```

### Q12 : L'anneau des entiers

```ocaml
let int_anneau = {
  addition = ( + );
  multiplication = ( * );
  zero = 0;
  one = 1;
  equal = ( = );
}
```

### Q13 : Evaluation generique

```ocaml
(* eval_expr : 'a anneau -> (string -> 'a) -> 'a expr -> 'a *)
let rec eval_expr an f e = match e with
  | Cst c -> c
  | Var s -> f s
  | Add (p, q) -> an.addition (eval_expr an f p) (eval_expr an f q)
  | Mul (p, q) -> an.multiplication (eval_expr an f p) (eval_expr an f q)
```

**Type** : Le record `anneau` permet d'utiliser la meme fonction `eval_expr` pour differents types numeriques (entiers, flottants, matrices, etc.).

### Q15 : Simplification

```ocaml
let rec simpl_expr e = match e with
  | Cst c -> Cst c
  | Var s -> Var s
  | Add (p, q) ->
      (match (p, q) with
       | (Cst 0, _) -> simpl_expr q      (* 0 + q = q *)
       | (_, Cst 0) -> simpl_expr p      (* p + 0 = p *)
       | (Var s1, Var s2) when s1 = s2 -> (* x + x = 2x *)
           Mul (Cst 2, Var s1)
       | (a, b) -> Add (simpl_expr a, simpl_expr b))
  | Mul (p, q) ->
      (match (p, q) with
       | (Cst 1, _) -> simpl_expr q      (* 1 * q = q *)
       | (_, Cst 1) -> simpl_expr p      (* p * 1 = p *)
       | (Cst 0, _) | (_, Cst 0) -> Cst 0  (* 0 * _ = 0 *)
       | (a, b) -> Mul (simpl_expr a, simpl_expr b))
```

### Q16-Q17 : Conversion linexpr <-> expr

```ocaml
(* expr -> linexpr *)
let rec linexpr_of_expr e = match e with
  | Cst c -> {cst = c; coeffs = []}
  | Var v -> {cst = 0; coeffs = [(1, v)]}
  | Add (e1, e2) ->
      let a = linexpr_of_expr e1 in
      let b = linexpr_of_expr e2 in
      normalise {cst = a.cst + b.cst; coeffs = a.coeffs @ b.coeffs}
  | Mul (Cst c, Var v) | Mul (Var v, Cst c) ->
      {cst = 0; coeffs = [(c, v)]}
  | _ -> failwith "Expression non lineaire"
```

**Point cle** : la conversion `Mul -> linexpr` n'est possible que pour `Cst * Var`. Toute autre multiplication (comme `Var * Var`) n'est pas lineaire.
