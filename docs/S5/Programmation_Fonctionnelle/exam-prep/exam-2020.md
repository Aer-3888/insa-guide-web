# Examen 2020 -- Representation binaire et ensembles

## Partie 1 : Nombres binaires

### Types

```ocaml
type bit = B0 | B1
type bint = bit list
(* [B1; B0; B1] represente 5 en binaire (bit de poids faible en tete) *)
(* B1 + B0*2 + B1*4 = 1 + 0 + 4 = 5 *)
```

### Q2 : Conversion bit -> int

```ocaml
(* int_of_bit : bit -> int *)
let int_of_bit b = match b with
  | B0 -> 0
  | B1 -> 1
```

### Q3 : Conversion bint -> int (avec fold_right)

```ocaml
(* int_of_bint : bint -> int *)
let int_of_bint l =
  List.fold_right (fun x y -> int_of_bit x + 2 * y) l 0
```

**Trace : `int_of_bint [B1; B0; B1]`**
```
fold_right f [B1; B0; B1] 0
= f B1 (f B0 (f B1 0))
= f B1 (f B0 (1 + 2*0))
= f B1 (f B0 1)
= f B1 (0 + 2*1)
= f B1 2
= 1 + 2*2 = 5
```

**Pourquoi fold_right ?** Le bit de poids faible est en tete de liste. fold_right traite d'abord le dernier element (poids fort), multipliant progressivement par 2.

### Q4-Q5 : Compter et supprimer les zeros

```ocaml
(* count_zeros : bint -> int *)
(* Compte les zeros en tete *)
let rec count_zeros l = match l with
  | [] -> 0
  | B0 :: rest -> 1 + count_zeros rest
  | B1 :: _ -> 0

(* count_and_remove : bint -> bint * int *)
let rec count_and_remove l = match l with
  | [] -> ([], 0)
  | B1 :: rest -> (B1 :: rest, 0)
  | B0 :: rest ->
      let (r, n) = count_and_remove rest in
      (r, n + 1)
```

### Q6 : Normalisation (supprimer les zeros de poids fort)

```ocaml
(* normalise : bint -> bint *)
let normalise l =
  List.rev (fst (count_and_remove (List.rev l)))
```

**Principe** : inverser, supprimer les zeros en tete (= zeros de poids fort), re-inverser.

**Trace : `normalise [B0; B1; B1; B0; B0]`**
```
List.rev [B0; B1; B1; B0; B0] = [B0; B0; B1; B1; B0]
count_and_remove [B0; B0; B1; B1; B0] = ([B1; B1; B0], 2)
fst (...) = [B1; B1; B0]
List.rev [B1; B1; B0] = [B0; B1; B1]
```

## Partie 2 : Ensemble trie

### Q7-Q9 : Tri et union

```ocaml
type comparison = EQUAL | LESSTHAN | GREATERTHAN

(* is_sorted : ('a -> 'a -> comparison) -> 'a list -> bool *)
let rec is_sorted cmp l = match l with
  | [] | [_] -> true
  | a :: b :: rest ->
      cmp a b <> GREATERTHAN && is_sorted cmp (b :: rest)

(* add_elt : ('a -> 'a -> comparison) -> 'a -> 'a list -> 'a list *)
let rec add_elt cmp e l = match l with
  | [] -> [e]
  | e1 :: rest ->
      if cmp e e1 = GREATERTHAN then e1 :: (add_elt cmp e rest)
      else e :: e1 :: rest

(* union : ('a -> 'a -> comparison) -> 'a list -> 'a list -> 'a list *)
let rec union cmp l1 l2 = match (l1, l2) with
  | ([], _) -> l2
  | (_, []) -> l1
  | (e1 :: r1, e2 :: r2) ->
      match cmp e1 e2 with
      | LESSTHAN -> e1 :: (union cmp r1 l2)
      | EQUAL -> e1 :: (union cmp r1 r2)  (* pas de doublon *)
      | GREATERTHAN -> e2 :: (union cmp l1 r2)
```

## Partie 3 : Ensembles d'entiers par arbre binaire

### Structure

```ocaml
type intset = Empty | Node of intset * bool * intset
(* Node (gauche, present, droite) *)
(* - present : l'entier [] est dans l'ensemble *)
(* - gauche : sous-arbre pour bit B0 *)
(* - droite : sous-arbre pour bit B1 *)
```

L'ensemble `{0; 2; 3; 4; 7}` en binaire :
- `0 = []`, `2 = [B0;B1]`, `3 = [B1;B1]`, `4 = [B0;B0;B1]`, `7 = [B1;B1;B1]`

### Q10-Q11 : Cardinal et appartenance

```ocaml
(* cardinal : intset -> int *)
let rec cardinal s = match s with
  | Empty -> 0
  | Node (g, true, d) -> 1 + cardinal g + cardinal d
  | Node (g, false, d) -> cardinal g + cardinal d

(* mem : intset -> bint -> bool *)
let rec mem s x = match s, x with
  | Empty, _ -> false
  | Node (_, b, _), [] -> b
  | Node (g, _, d), e :: rest ->
      if e = B0 then mem g rest else mem d rest
```

**Trace : `mem a [B0; B1]`** (cherche 2)
```
a = Node(gauche, true, droite)
B0 -> mem gauche [B1]
gauche = Node(gg, false, gd)
B1 -> mem gd []
gd = Node(_, true, _)
[] -> true
```

### Q12-Q14 : Singleton, ajout, suppression

```ocaml
(* singleton : bint -> intset *)
let rec singleton l = match l with
  | [] -> Node (Empty, true, Empty)
  | B0 :: rest -> Node (singleton rest, false, Empty)
  | B1 :: rest -> Node (Empty, false, singleton rest)

(* add_elt : bint -> intset -> intset *)
let rec add_elt i s = match i, s with
  | [], Empty -> Node (Empty, true, Empty)
  | [], Node (g, _, d) -> Node (g, true, d)
  | B0 :: rest, Empty -> Node (add_elt rest Empty, false, Empty)
  | B1 :: rest, Empty -> Node (Empty, false, add_elt rest Empty)
  | B0 :: rest, Node (g, b, d) -> Node (add_elt rest g, b, d)
  | B1 :: rest, Node (g, b, d) -> Node (g, b, add_elt rest d)

(* remove_elt : bint -> intset -> intset *)
let rec remove_elt i s = match i, s with
  | _, Empty -> Empty
  | [], Node (g, _, d) -> Node (g, false, d)
  | B0 :: rest, Node (g, b, d) -> Node (remove_elt rest g, b, d)
  | B1 :: rest, Node (g, b, d) -> Node (g, b, remove_elt rest d)
```

### Q17 : Union d'ensembles

```ocaml
(* union : intset -> intset -> intset *)
let rec union s1 s2 = match s1, s2 with
  | Empty, s | s, Empty -> s
  | Node (g1, b1, d1), Node (g2, b2, d2) ->
      Node (union g1 g2, b1 || b2, union d1 d2)
```

**Point cle** : `b1 || b2` -- un element est dans l'union s'il est dans l'un OU l'autre ensemble.

### Q18 : Division par 2

```ocaml
(* div2 : intset -> intset *)
(* {n/2 | n dans S, n pair} union {(n-1)/2 | n dans S, n impair} *)
let div2 s = match s with
  | Empty -> Empty
  | Node (g, b, d) -> union g d
```

**Explication** : diviser par 2 en binaire = decaler d'un bit vers la droite = supprimer le premier bit. Les nombres pairs (bit 0 = B0) sont dans le sous-arbre gauche, les impairs (bit 0 = B1) dans le droit. Apres division, ils se retrouvent au meme niveau, d'ou l'union.

### Q19 : Lister tous les elements

```ocaml
(* elements : intset -> bint list *)
let rec elements s = match s with
  | Empty -> []
  | Node (g, true, d) ->
      [[]] @ (List.map (fun i -> B0 :: i) (elements g))
           @ (List.map (fun i -> B1 :: i) (elements d))
  | Node (g, false, d) ->
      (List.map (fun i -> B0 :: i) (elements g))
    @ (List.map (fun i -> B1 :: i) (elements d))
```

**Principe** : on reconstruit les chemins. Si present=true, on ajoute `[]` (l'element courant). Pour chaque element du sous-arbre gauche, on prefixe par B0. Pour le droit, par B1.
